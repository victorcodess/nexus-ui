## Usage

```tsx keepBackground
import { toast } from "sonner";
import { Toaster } from "@/components/nexus-ui/toaster";
```

```tsx keepBackground
<Toaster />
```

```tsx keepBackground
toast.success("Saved");
toast.info("New update available");
toast.warning("Connection is unstable");
toast.error("Request failed");
toast.loading("Syncing...");
```
